-- COMANDO PARA CREAR UNA RAMA Y SALTAR A ELLA: git checkout -b develop
-- COMANDO PARA SALTAR A LA RAMA DE DESARROLLO / PRUEBAS: git checkout develop
-- COMANDO PARA SALTAR A LA RAMA MASTER: git checkout main
-- COMANDO PARA BORRAR UNA RAMA: git branch -d develop
-- COMANDO PARA INCORPORAR LOS CAMBIOS A LA RAMA MASTER Y PONERLOS EN PRODUCCION: git merge develop
-- COMANDO PARA CREAR RAMA REMOTA DE DESARROLLO: git push --set-upstream origin develop
-- COMANDO PARA ACTUALIZAR RAMA DE DESARROLLO / PRUEBAS: git pull origin develop
